  \textbf{Massa}
  \begin{itemize}
  \item Farinha de trigo
  \item Água
  \item Azeite
  \item Sal
  \end{itemize}

  \textbf{Recheio 1}
  \begin{itemize}
  \item Carne moída
  \item Queijo muçarela ou tipo feta
  \item Azeite
  \item Alho
  \item Cebola
  \item Pimentão vermelho
  \item Escarola ou chicória
  \item Paprica
  \item Sal
  \item Água
  \end{itemize}

  \textbf{Recheio 2}
  \begin{itemize}
  \item Escarola
  \item Ovo
  \item Queijo tipo feta
  \item Muçarela
  \item Alho
  \end{itemize}
}{
  \textbf{Massa}
  \begin{enumerate}
  \item Misturar tudo e repousar 30 minutos. Preparar os recheios nesse meio tempo. Só abrir a massa quando o
    recheio estiver frio.
  \item Abrir a massa em formato retangular com o rolo, espessura bem fina de 2-3 mm.
  \end{enumerate}

  \textbf{Recheio 1}
  \begin{enumerate}
  \item Refogar 125 g de carne moída com os temperos, sem o queijo, e deixar esfriar.\footnote{Se estiver
      úmida demais ou quente, vai furar a massa.}
  \item Espalhar a carne em uma camada fina sobre a massa, cobrir com queijo e escarola pré-cozida e
    espremida. Cobrir com a outra metade da massa.
  \item Umedecer a borda da massa para fechar bem.
  \item Transferir para uma frigideira grande em fogo médio/alto, untada com azeite, e deixar por $\sim$ 8 min
    de cada lado.
  \end{enumerate}

  \textbf{Recheio 2}
  \begin{enumerate}
  \item Misturar 10 folhas de escarola ou outro vegetal, picada, cozida, esfriada e espremida com o ovo, 1
    dente de alho, 30 g de queijo muçarela e 30 g de queijo tipo feta.
  \item Colocar o recheio em uma metade da massa e tampar com a outra metade. Fechar bem, umedecendo as
    bordas.
  \item Transferir para uma frigideira quente, untada com azeite, fogo médio/alto. Deixar de 5 a 8 minutos de
    cada lado.
  \end{enumerate}
}
