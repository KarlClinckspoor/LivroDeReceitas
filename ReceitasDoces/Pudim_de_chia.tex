\receitaemoji[\karlAprova \karenAprova]{Pudim de chia}{
  \textsc{Chocolate e manteiga de amendoim}
  \begin{itemize}
  \item 1,5 xícara de leite
  \item 2 colheres de chá de açúcar
  \item 2 colheres de chá de manteiga de amendoim
  \item 2 colheres de sopa de chocolate em pó
  \item 3 colheres de sopa de sementes de chia
  \end{itemize}

  \textsc{Coco e baunilha}
  \begin{itemize}
  \item 1,5 xícara de leite de coco ou leite comum
  \item 2 colheres de chá de açúcar
  \item 1 colher de chá de baunilha
  \item 3 colheres de sopa de sementes de chia
  \end{itemize}
}{
  \begin{enumerate}
    
  \item Misturar a chia com leite quente\footnote{Ajuda na digestão e esteriliza as sementes}
  \item Colocar o resto dos ingredientes e colocar na geladeira até o dia seguinte.
  \item No dia seguinte, misturar bem para ficar com consistência de pudim.
  \end{enumerate}
}
