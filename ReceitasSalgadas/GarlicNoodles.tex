\receitaemoji[\testado \karlAprova]{Garlic Noodles}{
	\textsc{Molho}
	\begin{itemize}
		\item 3/2 colher de sopa de shoyu
		\item 1 colher de sopa de molho de ostra
		\item 3/2 colher de chá de molho inglês Worchestershire
		\item 1 colher de chá de molho de peixe
		\item 1/4 colher de chá de óleo de gergelim
		\item Pitada de pimenta caiena
	\end{itemize}

	\textsc{Resto}
	\begin{itemize}
		\item 170 g de macarrão seco
		\item 4 colheres de sopa de manteiga sem sal
		\item 4 dentes de alho
		\item 1/4 xícara de parmesão ralado
	\end{itemize}
}{
	\begin{enumerate}
		\item Derreter a manteiga numa frigideira sob calor médio.%
      % \footnote{Ver
        % \verb{https://www.allrecipes.com/recipe/270770/garlic-noodles/} e
        % \texttt{https://www.youtube.com/watch?v=VZckD_bZlso}}
		\item Colocar o alho por 1 minuto. Não deixar dourar, é bem pouquinho tempo mesmo.
		\item Colocar os molhos e desligar o fogo.
		\item Cozinhar o macarrão.
		\item Colocar o macarrão no molho utilizando um pegador. É bom que venha com um pouco de água mesmo.
		\item Recobrir o macarrão com o molho mexendo com o pegador.
		\item Colocar o queijo. Se ficar seco, colocar um pouco da água de cozimento.
		\item Servir.
	\end{enumerate}
}
